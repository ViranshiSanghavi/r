import { Link } from 'react-router-dom'// link is like <a> tag in html. instead of the whoel page relaod, it sends

//it tells the BrowserRouter to simply update the URL in the address bar

 

export const Home = () => <h2>Home Page</h2>;

export const About = () => <h2>About Page</h2>;

export const Contact = () => <h2>Contact Page</h2>;

 

function Nav() {

  return (

    <nav style={{ marginRight: '20px', width: '200px' }}>

      <ul style={{ listStyleType: 'none', padding: 0 }}>

        <li><Link to="/">Home</Link></li>

        <li><Link to="/about">About</Link></li>

        <li><Link to="/contact">Contact</Link></li>

      </ul>

    </nav>

  )

}

 

export default Nav
