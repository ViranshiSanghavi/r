import { useState } from 'react'
import reactLogofrom './assets/react.svg'
import viteLogofrom '/vite.svg'
import './App.css'
import {useEffect } from "react";
 
function App() {
 const [count, setCount] = useState(0)
 const [tasks, setTasks] = useState([]);  //This state for maintaining task list 
 const [text, setText] = useState("");   //This state for recording what user is typing
 const [nextId, setNextId] = useState(1);
 
  useEffect(() => {                     //This is hook in react. It runs everytime compoenent re-renders. ie. only when tasks are added
 }, [tasks]);
 
 constaddTask = () => {       // This function is called when button is clicked to add task
   if (text.trim()==="") return;
   //... here is spread operator. It takes existing items of task array
   //and appends new task items with new id, changed state text. 
   setTasks([...tasks,{ id: nextId, text: text, selected: false}]); //It updates task array with new task in this format
   setNextId(nextId+ 1);
   setText("");      //To clear the teaxt in the field
 };
  constselectTask = (id) => {       // selects or unselects a task when the checkbox is clicked.
   setTasks(
     tasks.map(task=>               //Goes through every task in the list
       task.id=== id                //mapping teh id to taks id
         ? { ...task, selected: !task.selected }   //if yes , change teh status of chekcbox. 
         :task
     )
   );
 };
  constdeleteTask = () => {         //This fucntion is called when user clicks delete button. 
   setTasks(tasks.filter(task=> !task.selected)); // filter() goes through every task. Keeps only tasks whose id is NOT equal to the clicked task’s id
 };
 
 return (
   <>
    
     <h1>To do list</h1>
       <div>
       <input
         type="text"
         placeholder="Enter a task"
         value={text}
         onChange={(e) => setText(e.target.value)}
       />
       <buttononClick={addTask}>Add</button>
       <buttononClick={deleteTask}>Delete</button>
       <ul>
       {tasks.map(task=> (
         <likey={task.id}>
           <input
             type="checkbox"
             checked={task.selected}
             onChange={() => selectTask(task.id)}
           />
 
           <span>{task.text}</span>
         </li>
       ))}
     </ul>
     </div>
     
     <div>
       <ahref="https://vite.dev"target="_blank">
         <imgsrc={viteLogo}className="logo"alt="Vite logo" />
       </a>
       <ahref="https://react.dev"target="_blank">
         <imgsrc={reactLogo}className="logo react" alt="React logo" />
       </a>
     </div>
     <h1>Vite + React</h1>
     <divclassName="card">
       <buttononClick={() => setCount((count) => count + 1)}>
         count is {count}
       </button>
       <p>
         Edit <code>src/App.jsx</code> and save to test HMR
       </p>
     </div>
     <pclassName="read-the-docs">
       Click on the Vite and React logos to learn more
     </p>
   </>
 )
}
 
export defaultApp
